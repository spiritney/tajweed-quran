"__BASMALAH__\n__SURAH_17__\nسُبْحَٰنَ اَ۬لذِے أَسْرَيٰ بِعَبْدِهِۦ لَيْلاٗ مِّنَ اَ۬لْمَسْجِدِ اِ۬لْحَرَامِ إِلَي\nاَ۬لْمَسْجِدِ اِ۬لْأَقْصَا اَ۬لذِے بَٰرَكْنَا حَوْلَهُۥ لِنُرِيَهُۥ مِنْ ءَايَٰتِنَاۖ إِنَّهُۥ\nهُوَ اَ۬لسَّمِيعُ اُ۬لْبَصِيرُۖ (1) وَءَاتَيْنَا مُوسَي اَ۬لْكِتَٰبَ وَجَعَلْنَٰهُ\nهُديٗ لِّبَنِے إِسْرَآءِيلَ أَلَّا تَتَّخِذُواْ مِن دُونِے وَكِيلاٗ (2)\nذُرِّيَّةَ مَنْ حَمَلْنَا مَعَ نُوحٍۖ إِنَّهُۥ كَانَ عَبْداٗ شَكُوراٗۖ (3)\nوَقَضَيْنَا إِلَيٰ بَنِے إِسْرَآءِيلَ فِے اِ۬لْكِتَٰبِ لَتُفْسِدُنَّ فِے اِ۬لْأَرْضِ\nمَرَّتَيْنِ وَلَتَعْلُنَّ عُلُوّاٗ كَبِيراٗۖ (4) فَإِذَا جَآءَ وَعْدُ أُولَيٰهُمَا\nبَعَثْنَا عَلَيْكُمْ عِبَاداٗ لَّنَا أُوْلِے بَأْسٖ شَدِيدٖ فَجَاسُواْ خِلَٰلَ\nاَ۬لدِّيَارِۖ وَكَانَ وَعْداٗ مَّفْعُولاٗۖ (5) ثُمَّ رَدَدْنَا لَكُمُ اُ۬لْكَرَّةَ\nعَلَيْهِمْ وَأَمْدَدْنَٰكُم بِأَمْوَٰلٖ وَبَنِينَ وَجَعَلْنَٰكُمْ أَكْثَرَ نَفِيراًۖ\n(6) إِنْ أَحْسَنتُمْ أَحْسَنتُمْ لِأَنفُسِكُمْۖ وَإِنْ أَسَأْتُمْ فَلَهَاۖ فَإِذَا\nجَآءَ وَعْدُ اُ۬لْأٓخِرَةِ لِيَسُـُٔواْ وُجُوهَكُمْ وَلِيَدْخُلُواْ اُ۬لْمَسْجِدَ\nكَمَا دَخَلُوهُ أَوَّلَ مَرَّةٖ وَلِيُتَبِّرُواْ مَا عَلَوْاْ تَتْبِيراًۖ (7)\n\nعَسَيٰ رَبُّكُمْ أَنْ يَّرْحَمَكُمْۖ وَإِنْ عُدتُّمْ عُدْنَاۖ وَجَعَلْنَا جَهَنَّمَ لِلْكَٰفِرِينَ\nحَصِيراًۖ (8) إِنَّ هَٰذَا اَ۬لْقُرْءَانَ يَهْدِے لِلتِے هِيَ أَقْوَمُ وَيُبَشِّرُ\nاُ۬لْمُؤْمِنِينَ اَ۬لذِينَ يَعْمَلُونَ اَ۬لصَّٰلِحَٰتِ أَنَّ لَهُمْ أَجْراٗ كَبِيراٗ (9)\nوَأَنَّ اَ۬لذِينَ لَا يُؤْمِنُونَ بِالْأٓخِرَةِ أَعْتَدْنَا لَهُمْ عَذَاباً أَلِيماٗۖ (10)\n۞وَيَدْعُ اُ۬لْإِنسَٰنُ بِالشَّرِّ دُعَآءَهُۥ بِالْخَيْرِۖ وَكَانَ اَ۬لْإِنسَٰنُ عَجُولاٗۖ (11)\nوَجَعَلْنَا اَ۬ليْلَ وَالنَّهَارَ ءَايَتَيْنِ فَمَحَوْنَا ءَايَةَ اَ۬ليْلِ وَجَعَلْنَا ءَايَةَ\nاَ۬لنَّهَارِ مُبْصِرَةٗ لِّتَبْتَغُواْ فَضْلاٗ مِّن رَّبِّكُمْ وَلِتَعْلَمُواْ عَدَدَ\nاَ۬لسِّنِينَ وَالْحِسَابَۖ وَكُلَّ شَےْءٖ فَصَّلْنَٰهُ تَفْصِيلاٗۖ (12) وَكُلَّ\nإِنسَٰنٍ أَلْزَمْنَٰهُ طَٰٓئِرَهُۥ فِے عُنُقِهِۦ وَنُخْرِجُ لَهُۥ يَوْمَ اَ۬لْقِيَٰمَةِ كِتَٰباٗ\nيَلْقَيٰهُ مَنشُوراًۖ (13) اِ۪قْرَأْ كِتَٰبَكَ كَفَيٰ بِنَفْسِكَ اَ۬لْيَوْمَ عَلَيْكَ حَسِيباٗۖ\n(14) مَّنِ اِ۪هْتَدَيٰ فَإِنَّمَا يَهْتَدِے لِنَفْسِهِۦۖ وَمَن ضَلَّ فَإِنَّمَا يَضِلُّ\nعَلَيْهَاۖ وَلَا تَزِرُ وَازِرَةٞ وِزْرَ أُخْرَيٰۖ وَمَا كُنَّا مُعَذِّبِينَ حَتَّيٰ نَبْعَثَ\nرَسُولاٗۖ (15) وَإِذَا أَرَدْنَا أَن نُّهْلِكَ قَرْيَةً أَمَرْنَا مُتْرَفِيهَا فَفَسَقُواْ فِيهَا\nفَحَقَّ عَلَيْهَا اَ۬لْقَوْلُ فَدَمَّرْنَٰهَا تَدْمِيراٗۖ (16) وَكَمْ أَهْلَكْنَا مِنَ اَ۬لْقُرُونِ\nمِنۢ بَعْدِ نُوحٖۖ وَكَفَيٰ بِرَبِّكَ بِذُنُوبِ عِبَادِهِۦ خَبِيراَۢ بَصِيراٗۖ (17)\n\nمَّن كَانَ يُرِيدُ اُ۬لْعَاجِلَةَ عَجَّلْنَا لَهُۥ فِيهَا مَا نَشَآءُ لِمَن نُّرِيدُ ثُمَّ\nجَعَلْنَا لَهُۥ جَهَنَّمَ يَصْلَيٰهَا مَذْمُوماٗ مَّدْحُوراٗۖ (18) وَمَنْ أَرَادَ\nاَ۬لْأٓخِرَةَ وَسَعَيٰ لَهَا سَعْيَهَا وَهْوَ مُؤْمِنٞ فَأُوْلَٰٓئِكَ كَانَ\nسَعْيُهُم مَّشْكُوراٗۖ (19) كُلّاٗ نُّمِدُّ هَٰؤُلَآءِ وَهَٰؤُلَآءِ مِنْ\nعَطَآءِ رَبِّكَۖ وَمَا كَانَ عَطَآءُ رَبِّكَ مَحْظُوراًۖ (20) اُ۟نظُرْ كَيْفَ\nفَضَّلْنَا بَعْضَهُمْ عَلَيٰ بَعْضٖۖ وَلَلْأٓخِرَةُ أَكْبَرُ دَرَجَٰتٖ وَأَكْبَرُ\nتَفْضِيلاٗۖ (21) لَّا تَجْعَلْ مَعَ اَ۬للَّهِ إِلَٰهاً ءَاخَرَ فَتَقْعُدَ مَذْمُوماٗ مَّخْذُولاٗۖ\n(22) ۞وَقَضَيٰ رَبُّكَ أَلَّا تَعْبُدُواْ إِلَّا إِيَّاهُ وَبِالْوَٰلِدَيْنِ إِحْسَٰناًۖ إِمَّا\nيَبْلُغَنَّ عِندَكَ اَ۬لْكِبَرَ أَحَدُهُمَا أَوْ كِلَاهُمَا فَلَا تَقُل لَّهُمَا\nأُفّٖ وَلَا تَنْهَرْهُمَا وَقُل لَّهُمَا قَوْلاٗ كَرِيماٗۖ (23) وَاخْفِضْ لَهُمَا\nجَنَاحَ اَ۬لذُّلِّ مِنَ اَ۬لرَّحْمَةِ وَقُل رَّبِّ اِ۪رْحَمْهُمَا كَمَا رَبَّيَٰنِے\nصَغِيراٗۖ (24) رَّبُّكُمْ أَعْلَمُ بِمَا فِے نُفُوسِكُمْۖ إِن تَكُونُواْ صَٰلِحِينَ\nفَإِنَّهُۥ كَانَ لِلْأَوَّٰبِينَ غَفُوراٗۖ (25) وَءَاتِ ذَا اَ۬لْقُرْبَيٰ حَقَّهُۥ\nوَالْمِسْكِينَ وَابْنَ اَ۬لسَّبِيلِ وَلَا تُبَذِّرْ تَبْذِيراًۖ (26) إِنَّ اَ۬لْمُبَذِّرِينَ\nكَانُواْ إِخْوَٰنَ اَ۬لشَّيَٰطِينِۖ وَكَانَ اَ۬لشَّيْطَٰنُ لِرَبِّهِۦ كَفُوراٗۖ (27)\n\nوَإِمَّا تُعْرِضَنَّ عَنْهُمُ اُ۪بْتِغَآءَ رَحْمَةٖ مِّن رَّبِّكَ تَرْجُوهَا فَقُل لَّهُمْ قَوْلاٗ\nمَّيْسُوراٗۖ (28) وَلَا تَجْعَلْ يَدَكَ مَغْلُولَةً إِلَيٰ عُنُقِكَ وَلَا تَبْسُطْهَا\nكُلَّ اَ۬لْبَسْطِ فَتَقْعُدَ مَلُوماٗ مَّحْسُوراًۖ (29) إِنَّ رَبَّكَ يَبْسُطُ اُ۬لرِّزْقَ\nلِمَنْ يَّشَآءُ وَيَقْدِرُۖ إِنَّهُۥ كَانَ بِعِبَادِهِۦ خَبِيراَۢ بَصِيراٗۖ (30) وَلَا تَقْتُلُواْ\nأَوْلَٰدَكُمْ خَشْيَةَ إِمْلَٰقٖۖ نَّحْنُ نَرْزُقُهُمْ وَإِيَّاكُمْۖ إِنَّ قَتْلَهُمْ كَانَ\nخِطْـٔاٗ كَبِيراٗۖ (31) وَلَا تَقْرَبُواْ اُ۬لزِّنَيٰ إِنَّهُۥ كَانَ فَٰحِشَةٗ وَسَآءَ\nسَبِيلاٗۖ (32) وَلَا تَقْتُلُواْ اُ۬لنَّفْسَ اَ۬لتِے حَرَّمَ اَ۬للَّهُ إِلَّا بِالْحَقِّۖ\nوَمَن قُتِلَ مَظْلُوماٗ فَقَدْ جَعَلْنَا لِوَلِيِّهِۦ سُلْطَٰناٗ فَلَا يُسْرِف فِّے\nاِ۬لْقَتْلِۖ إِنَّهُۥ كَانَ مَنصُوراٗۖ (33) وَلَا تَقْرَبُواْ مَالَ اَ۬لْيَتِيمِ إِلَّا بِالتِے\nهِيَ أَحْسَنُ حَتَّيٰ يَبْلُغَ أَشُدَّهُۥۖ وَأَوْفُواْ بِالْعَهْدِ إِنَّ اَ۬لْعَهْدَ كَانَ\nمَسْـُٔولاٗۖ (34) وَأَوْفُواْ اُ۬لْكَيْلَ إِذَا كِلْتُمْ وَزِنُواْ بِالْقُسْطَاسِ اِ۬لْمُسْتَقِيمِۖ\nذَٰلِكَ خَيْرٞ وَأَحْسَنُ تَأْوِيلاٗۖ (35) ۞وَلَا تَقْفُ مَا لَيْسَ لَكَ بِهِۦ عِلْمٌۖ إِنَّ\nاَ۬لسَّمْعَ وَالْبَصَرَ وَالْفُؤَادَ كُلُّ أُوْلَٰٓئِكَ كَانَ عَنْهُ مَسْـُٔولاٗۖ (36)\nوَلَا تَمْشِ فِے اِ۬لْأَرْضِ مَرَحاًۖ إِنَّكَ لَن تَخْرِقَ اَ۬لْأَرْضَ وَلَن تَبْلُغَ\nاَ۬لْجِبَالَ طُولاٗۖ (37) كُلُّ ذَٰلِكَ كَانَ سَيِّئَةً عِندَ رَبِّكَ مَكْرُوهاٗۖ (38)\n\nذَٰلِكَ مِمَّا أَوْحَيٰ إِلَيْكَ رَبُّكَ مِنَ اَ۬لْحِكْمَةِۖ وَلَا تَجْعَلْ مَعَ اَ۬للَّهِ إِلَٰهاً\nءَاخَرَ فَتُلْقَيٰ فِے جَهَنَّمَ مَلُوماٗ مَّدْحُوراًۖ (39) أَفَأَصْفَيٰكُمْ رَبُّكُم\nبِالْبَنِينَ وَاتَّخَذَ مِنَ اَ۬لْمَلَٰٓئِكَةِ إِنَٰثاًۖ إِنَّكُمْ لَتَقُولُونَ قَوْلاً عَظِيماٗۖ (40)\nوَلَقَدْ صَرَّفْنَا فِے هَٰذَا اَ۬لْقُرْءَانِ لِيَذَّكَّرُواْ وَمَا يَزِيدُهُمْ إِلَّا نُفُوراٗۖ (41)\nقُل لَّوْ كَانَ مَعَهُۥ ءَالِهَةٞ كَمَا تَقُولُونَ إِذاٗ لَّابْتَغَوْاْ إِلَيٰ ذِے اِ۬لْعَرْشِ سَبِيلاٗۖ\n(42) سُبْحَٰنَهُۥ وَتَعَٰلَيٰ عَمَّا يَقُولُونَ عُلُوّاٗ كَبِيراٗۖ (43) يُسَبِّحُ لَهُ اُ۬لسَّمَٰوَٰتُ\nاُ۬لسَّبْعُ وَالْأَرْضُ وَمَن فِيهِنَّۖ وَإِن مِّن شَےْءٍ إِلَّا يُسَبِّحُ بِحَمْدِهِۦ وَلَٰكِن\nلَّا تَفْقَهُونَ تَسْبِيحَهُمْۖ إِنَّهُۥ كَانَ حَلِيماً غَفُوراٗۖ (44) وَإِذَا قَرَأْتَ\nاَ۬لْقُرْءَانَ جَعَلْنَا بَيْنَكَ وَبَيْنَ اَ۬لذِينَ لَا يُؤْمِنُونَ بِالْأٓخِرَةِ حِجَاباٗ\nمَّسْتُوراٗۖ (45) وَجَعَلْنَا عَلَيٰ قُلُوبِهِمْ أَكِنَّةً أَنْ يَّفْقَهُوهُ وَفِے ءَاذَانِهِمْ\nوَقْراٗۖ وَإِذَا ذَكَرْتَ رَبَّكَ فِے اِ۬لْقُرْءَانِ وَحْدَهُۥ وَلَّوْاْ عَلَيٰ أَدْبَٰرِهِمْ نُفُوراٗۖ (46)\nنَّحْنُ أَعْلَمُ بِمَا يَسْتَمِعُونَ بِهِۦ إِذْ يَسْتَمِعُونَ إِلَيْكَ وَإِذْ هُمْ نَجْوَيٰ\nإِذْ يَقُولُ اُ۬لظَّٰلِمُونَ إِن تَتَّبِعُونَ إِلَّا رَجُلاٗ مَّسْحُوراًۖ (47) اُ۟نظُرْ\nكَيْفَ ضَرَبُواْ لَكَ اَ۬لْأَمْثَالَ فَضَلُّواْ فَلَا يَسْتَطِيعُونَ سَبِيلاٗۖ (48)\nوَقَالُواْ أَٰ۟ذَا كُنَّا عِظَٰماٗ وَرُفَٰتاً إِنَّا لَمَبْعُوثُونَ خَلْقاٗ جَدِيداٗۖ (49)\n\n۞قُلْ وَبُكْماٗ\nوَصُمّاٗۖ مَّأْوَيٰهُمْ جَهَنَّمُۖ كُلَّمَا خَبَتْ زِدْنَٰهُمْ سَعِيراٗۖ (97)\nذَٰلِكَ جَزَآؤُهُم بِأَنَّهُمْ كَفَرُواْ بِـَٔايَٰتِنَا وَقَالُواْ أَٰ۟ذَا كُنَّا عِظَٰماٗ\nوَرُفَٰتاً إِنَّا لَمَبْعُوثُونَ خَلْقاٗ جَدِيداًۖ (98) ۞أَوَلَمْ يَرَوْاْ أَنَّ اَ۬للَّهَ\nاَ۬لذِے خَلَقَ اَ۬لسَّمَٰوَٰتِ وَالْأَرْضَ قَادِرٌ عَلَيٰ أَنْ يَّخْلُقَ مِثْلَهُمْ\nوَجَعَلَ لَهُمْ أَجَلاٗ لَّا رَيْبَ فِيهِ فَأَبَي اَ۬لظَّٰلِمُونَ إِلَّا كُفُوراٗۖ (99)\nقُل لَّوْ أَنتُمْ تَمْلِكُونَ خَزَآئِنَ رَحْمَةِ رَبِّيَ إِذاٗ لَّأَمْسَكْتُمْ خَشْيَةَ\nاَ۬لْإِنفَاقِۖ وَكَانَ اَ۬لْإِنسَٰنُ قَتُوراٗۖ (100) وَلَقَدْ ءَاتَيْنَا مُوسَيٰ تِسْعَ\nءَايَٰتِۢ بَيِّنَٰتٖۖ فَسْـَٔلْ بَنِے إِسْرَآءِيلَ إِذْ جَآءَهُمْ فَقَالَ لَهُۥ فِرْعَوْنُ\nإِنِّے لَأَظُنُّكَ يَٰمُوسَيٰ مَسْحُوراٗۖ (101) قَالَ لَقَدْ عَلِمْتَ مَا أَنزَلَ\nهَٰؤُلَآ۟ إِلَّا رَبُّ اُ۬لسَّمَٰوَٰتِ وَالْأَرْضِ بَصَآئِرَۖ وَإِنِّے لَأَظُنُّكَ\nيَٰفِرْعَوْنُ مَثْبُوراٗۖ (102) فَأَرَادَ أَنْ يَّسْتَفِزَّهُم مِّنَ اَ۬لْأَرْضِ\nفَأَغْرَقْنَٰهُ وَمَن مَّعَهُۥ جَمِيعاٗۖ (103) وَقُلْنَا مِنۢ بَعْدِهِۦ لِبَنِے إِسْرَآءِيلَ\nاَ۟سْكُنُواْ اُ۬لْأَرْضَۖ فَإِذَا جَآءَ وَعْدُ اُ۬لْأٓخِرَةِ جِئْنَا بِكُمْ لَفِيفاٗۖ (104)\n\nوَبِالْحَقِّ أَنزَلْنَٰهُ وَبِالْحَقِّ نَزَلَۖ وَمَا أَرْسَلْنَٰكَ إِلَّا مُبَشِّراٗ وَنَذِيراٗۖ (105)\nوَقُرْءَاناٗ فَرَقْنَٰهُ لِتَقْرَأَهُۥ عَلَي اَ۬لنَّاسِ عَلَيٰ مُكْثٖ وَنَزَّلْنَٰهُ تَنزِيلاٗۖ (106)\nقُلْ ءَامِنُواْ بِهِۦ أَوْ لَا تُؤْمِنُواْۖ إِنَّ اَ۬لذِينَ أُوتُواْ اُ۬لْعِلْمَ مِن قَبْلِهِۦ إِذَا يُتْلَيٰ\nعَلَيْهِمْ يَخِرُّونَ لِلْأَذْقَانِ سُجَّداٗ وَيَقُولُونَ سُبْحَٰنَ رَبِّنَا إِن كَانَ\nوَعْدُ رَبِّنَا لَمَفْعُولاٗۖ (107) وَيَخِرُّونَ لِلْأَذْقَانِ يَبْكُونَ وَيَزِيدُهُمْ\nخُشُوعاٗۖ۩ (108) قُلُ اُ۟دْعُواْ اُ۬للَّهَ أَوُ اُ۟دْعُواْ اُ۬لرَّحْمَٰنَۖ أَيّاٗ مَّا تَدْعُواْ فَلَهُ\nاُ۬لْأَسْمَآءُ اُ۬لْحُسْنَيٰۖ وَلَا تَجْهَرْ بِصَلَاتِكَ وَلَا تُخَافِتْ بِهَا وَابْتَغِ\nبَيْنَ ذَٰلِكَ سَبِيلاٗۖ (109) وَقُلِ اِ۬لْحَمْدُ لِلهِ اِ۬لذِے لَمْ يَتَّخِذْ وَلَداٗ وَلَمْ يَكُن\nلَّهُۥ شَرِيكٞ فِے اِ۬لْمُلْكِ وَلَمْ يَكُن لَّهُۥ وَلِيّٞ مِّنَ اَ۬لذُّلِّۖ وَكَبِّرْهُ تَكْبِيراَۢۖ (110)"
