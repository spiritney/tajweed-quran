"__BASMALAH__\n__SURAH_14__\nأَلَٓرَۖ كِتَٰبٌ أَنزَلْنَٰهُ إِلَيْكَ لِتُخْرِجَ اَ۬لنَّاسَ مِنَ اَ۬لظُّلُمَٰتِ\nإِلَي اَ۬لنُّورِ (1) بِإِذْنِ رَبِّهِمْ إِلَيٰ صِرَٰطِ اِ۬لْعَزِيزِ اِ۬لْحَمِيدِۖ (2)\nاِ۬للَّهُ اُ۬لذِے لَهُۥ مَا فِے اِ۬لسَّمَٰوَٰتِ وَمَا فِے اِ۬لْأَرْضِۖ وَوَيْلٞ\nلِّلْكَٰفِرِينَ مِنْ عَذَابٖ شَدِيدٍۖ (3) اِ۬لذِينَ يَسْتَحِبُّونَ\nاَ۬لْحَيَوٰةَ اَ۬لدُّنْيَا عَلَي اَ۬لْأٓخِرَةِ وَيَصُدُّونَ عَن سَبِيلِ\nاِ۬للَّهِ وَيَبْغُونَهَا عِوَجاً أُوْلَٰٓئِكَ فِے ضَلَٰلِۢ بَعِيدٖۖ (4) وَمَا\nأَرْسَلْنَا مِن رَّسُولٍ إِلَّا بِلِسَانِ قَوْمِهِۦ لِيُبَيِّنَ لَهُمْۖ\nفَيُضِلُّ اُ۬للَّهُ مَنْ يَّشَآءُ وَيَهْدِے مَنْ يَّشَآءُۖ وَهْوَ اَ۬لْعَزِيزُ\nاُ۬لْحَكِيمُۖ (5) وَلَقَدْ أَرْسَلْنَا مُوسَيٰ بِـَٔايَٰتِنَا أَنْ أَخْرِجْ\nقَوْمَكَ مِنَ اَ۬لظُّلُمَٰتِ إِلَي اَ۬لنُّورِ (6) وَذَكِّرْهُم بِأَييَّٰمِ\nاِ۬للَّهِۖ إِنَّ فِے ذَٰلِكَ لَأٓيَٰتٖ لِّكُلِّ صَبَّارٖ شَكُورٖۖ (7)\n\nوَإِذْ قَالَ مُوسَيٰ لِقَوْمِهِ اِ۟ذْكُرُواْ نِعْمَةَ اَ۬للَّهِ عَلَيْكُمْ\nإِذْ أَنجَيٰكُم مِّنْ ءَالِ فِرْعَوْنَ يَسُومُونَكُمْ سُوٓءَ اَ۬لْعَذَابِ\nوَيُذَبِّحُونَ أَبْنَآءَكُمْ وَيَسْتَحْيُونَ نِسَآءَكُمْۖ وَفِے\nذَٰلِكُم بَلَآءٞ مِّن رَّبِّكُمْ عَظِيمٞۖ (8) وَإِذْ تَأَذَّنَ رَبُّكُمْ\nلَئِن شَكَرْتُمْ لَأَزِيدَنَّكُمْ وَلَئِن كَفَرْتُمْ إِنَّ عَذَابِے\nلَشَدِيدٞۖ (9) وَقَالَ مُوسَيٰ إِن تَكْفُرُواْ أَنتُمْ وَمَن فِے\nاِ۬لْأَرْضِ جَمِيعاٗ فَإِنَّ اَ۬للَّهَ لَغَنِيٌّ حَمِيدٌۖ (10) أَلَمْ يَأْتِكُمْ نَبَؤُاْ\nاُ۬لذِينَ مِن قَبْلِكُمْ قَوْمِ نُوحٖ وَعَادٖ وَثَمُودَ (11) وَالذِينَ\nمِنۢ بَعْدِهِمْ لَا يَعْلَمُهُمْ إِلَّا اَ۬للَّهُۖ جَآءَتْهُمْ رُسُلُهُم\nبِالْبَيِّنَٰتِ فَرَدُّواْ أَيْدِيَهُمْ فِے أَفْوَٰهِهِمْ وَقَالُواْ إِنَّا كَفَرْنَا\nبِمَا أُرْسِلْتُم بِهِۦ وَإِنَّا لَفِے شَكّٖ مِّمَّا تَدْعُونَنَا إِلَيْهِ مُرِيبٖۖ (12)\n۞قَالَتْ رُسُلُهُمْ أَفِے اِ۬للَّهِ شَكّٞ فَاطِرِ اِ۬لسَّمَٰوَٰتِ وَالْأَرْضِ\nيَدْعُوكُمْ لِيَغْفِرَ لَكُم مِّن ذُنُوبِكُمْ وَيُؤَخِّرَكُمْ إِلَيٰ أَجَلٖ\nمُّسَمّيٗۖ قَالُواْ إِنْ أَنتُمْ إِلَّا بَشَرٞ مِّثْلُنَا تُرِيدُونَ أَن تَصُدُّونَا\nعَمَّا كَانَ يَعْبُدُ ءَابَآؤُنَا فَأْتُونَا بِسُلْطَٰنٖ مُّبِينٖۖ (13)\n\nقَالَتْ لَهُمْ رُسُلُهُمْ إِن نَّحْنُ إِلَّا بَشَرٞ مِّثْلُكُمْ وَلَٰكِنَّ اَ۬للَّهَ\nيَمُنُّ عَلَيٰ مَنْ يَّشَآءُ مِنْ عِبَادِهِۦۖ وَمَا كَانَ لَنَا أَن نَّأْتِيَكُم\nبِسُلْطَٰنٍ إِلَّا بِإِذْنِ اِ۬للَّهِۖ وَعَلَي اَ۬للَّهِ فَلْيَتَوَكَّلِ اِ۬لْمُؤْمِنُونَۖ\n(14) وَمَا لَنَا أَلَّا نَتَوَكَّلَ عَلَي اَ۬للَّهِ وَقَدْ هَدَيٰنَا سُبُلَنَاۖ وَلَنَصْبِرَنَّ\nعَلَيٰ مَا ءَاذَيْتُمُونَاۖ وَعَلَي اَ۬للَّهِ فَلْيَتَوَكَّلِ اِ۬لْمُتَوَكِّلُونَۖ\n(15) وَقَالَ اَ۬لذِينَ كَفَرُواْ لِرُسُلِهِمْ لَنُخْرِجَنَّكُم مِّنْ أَرْضِنَا\nأَوْ لَتَعُودُنَّ فِے مِلَّتِنَاۖ فَأَوْحَيٰ إِلَيْهِمْ رَبُّهُمْ لَنُهْلِكَنَّ\nاَ۬لظَّٰلِمِينَ (16) وَلَنُسْكِنَنَّكُمُ اُ۬لْأَرْضَ مِنۢ بَعْدِهِمْۖ\nذَٰلِكَ لِمَنْ خَافَ مَقَامِے وَخَافَ وَعِيدِۖ (17) وَاسْتَفْتَحُواْ\nوَخَابَ كُلُّ جَبَّارٍ عَنِيدٖ (18) مِّنْ وَّرَآئِهِۦ جَهَنَّمُ وَيُسْقَيٰ\nمِن مَّآءٖ صَدِيدٖ (19) يَتَجَرَّعُهُۥ وَلَا يَكَادُ يُسِيغُهُۥ وَيَأْتِيهِ\nاِ۬لْمَوْتُ مِن كُلِّ مَكَانٖ وَمَا هُوَ بِمَيِّتٖ وَمِنْ وَّرَآئِهِۦ\nعَذَابٌ غَلِيظٞۖ (20) مَّثَلُ اُ۬لذِينَ كَفَرُواْ بِرَبِّهِمْ أَعْمَٰلُهُمْ\nكَرَمَادٍ اِ۪شْتَدَّتْ بِهِ اِ۬لرِّيَٰحُ فِے يَوْمٍ عَاصِفٖ لَّا يَقْدِرُونَ\nمِمَّا كَسَبُواْ عَلَيٰ شَےْءٖۖ ذَٰلِكَ هُوَ اَ۬لضَّلَٰلُ اُ۬لْبَعِيدُۖ (21)\n\n۞أَلَمْ تَرَ أَنَّ اَ۬للَّهَ خَلَقَ اَ۬لسَّمَٰوَٰتِ وَالْأَرْضَ بِالْحَقِّۖ إِنْ يَّشَأْ\nيُذْهِبْكُمْ وَيَأْتِ بِخَلْقٖ جَدِيدٖۖ وَمَا ذَٰلِكَ عَلَي اَ۬للَّهِ بِعَزِيزٖۖ\n(22) وَبَرَزُواْ لِلهِ جَمِيعاٗ فَقَالَ اَ۬لضُّعَفَٰٓؤُاْ لِلذِينَ اَ۪سْتَكْبَرُواْ\nإِنَّا كُنَّا لَكُمْ تَبَعاٗ فَهَلْ أَنتُم مُّغْنُونَ عَنَّا مِنْ عَذَابِ اِ۬للَّهِ\nمِن شَےْءٖۖ قَالُواْ لَوْ هَدَيٰنَا اَ۬للَّهُ لَهَدَيْنَٰكُمْۖ سَوَآءٌ عَلَيْنَا\nأَجَزِعْنَا أَمْ صَبَرْنَا مَا لَنَا مِن مَّحِيصٖۖ (23) وَقَالَ اَ۬لشَّيْطَٰنُ لَمَّا\nقُضِيَ اَ۬لْأَمْرُ إِنَّ اَ۬للَّهَ وَعَدَكُمْ وَعْدَ اَ۬لْحَقِّ وَوَعَدتُّكُمْ\nفَأَخْلَفْتُكُمْ وَمَا كَانَ لِے عَلَيْكُم مِّن سُلْطَٰنٍ إِلَّا\nأَن دَعَوْتُكُمْ فَاسْتَجَبْتُمْ لِے فَلَا تَلُومُونِے وَلُومُواْ أَنفُسَكُم\nمَّا أَنَا بِمُصْرِخِكُمْ وَمَا أَنتُم بِمُصْرِخِيَّ إِنِّے كَفَرْتُ\nبِمَا أَشْرَكْتُمُونِ مِن قَبْلُۖ إِنَّ اَ۬لظَّٰلِمِينَ لَهُمْ عَذَابٌ\nأَلِيمٞۖ (24) وَأُدْخِلَ اَ۬لذِينَ ءَامَنُواْ وَعَمِلُواْ اُ۬لصَّٰلِحَٰتِ جَنَّٰتٖ\nتَجْرِے مِن تَحْتِهَا اَ۬لْأَنْهَٰرُ خَٰلِدِينَ فِيهَا بِإِذْنِ رَبِّهِمْۖ تَحِيَّتُهُمْ\nفِيهَا سَلَٰمٌۖ (25) أَلَمْ تَرَ كَيْفَ ضَرَبَ اَ۬للَّهُ مَثَلاٗ كَلِمَةٗ طَيِّبَةٗ\nكَشَجَرَةٖ طَيِّبَةٍ أَصْلُهَا ثَابِتٞ وَفَرْعُهَا فِے اِ۬لسَّمَآءِ (26)\n\nتُؤْتِے أُكْلَهَا كُلَّ حِينِۢ بِإِذْنِ رَبِّهَاۖ وَيَضْرِبُ اُ۬للَّهُ اُ۬لْأَمْثَالَ\nلِلنَّاسِ لَعَلَّهُمْ يَتَذَكَّرُونَۖ (27) وَمَثَلُ كَلِمَةٍ خَبِيثَةٖ\nكَشَجَرَةٍ خَبِيثَةٍ اُ۟جْتُثَّتْ مِن فَوْقِ اِ۬لْأَرْضِ مَا لَهَا مِن\nقَرَارٖۖ (28) يُثَبِّتُ اُ۬للَّهُ اُ۬لذِينَ ءَامَنُواْ بِالْقَوْلِ اِ۬لثَّابِتِ فِے اِ۬لْحَيَوٰةِ\nاِ۬لدُّنْيَا وَفِے اِ۬لْأٓخِرَةِۖ وَيُضِلُّ اُ۬للَّهُ اُ۬لظَّٰلِمِينَۖ وَيَفْعَلُ اُ۬للَّهُ\nمَا يَشَآءُۖ (29) ۞اَ۬لَمْ تَرَ إِلَي اَ۬لذِينَ بَدَّلُواْ نِعْمَتَ اَ۬للَّهِ كُفْراٗ\nوَأَحَلُّواْ قَوْمَهُمْ دَارَ اَ۬لْبَوَارِ (30) جَهَنَّمَ يَصْلَوْنَهَاۖ وَبِئْسَ\nاَ۬لْقَرَارُۖ (31) وَجَعَلُواْ لِلهِ أَندَاداٗ لِّيُضِلُّواْ عَن سَبِيلِهِۦۖ قُلْ\nتَمَتَّعُواْ فَإِنَّ مَصِيرَكُمْ إِلَي اَ۬لنَّارِۖ (32) قُل لِّعِبَادِيَ اَ۬لذِينَ\nءَامَنُواْ يُقِيمُواْ اُ۬لصَّلَوٰةَ وَيُنفِقُواْ مِمَّا رَزَقْنَٰهُمْ سِرّاٗ وَعَلَٰنِيَةٗ\nمِّن قَبْلِ أَنْ يَّأْتِيَ يَوْمٞ لَّا بَيْعٞ فِيهِ وَلَا خِلَٰلٌۖ (33) اِ۬للَّهُ اُ۬لذِے\nخَلَقَ اَ۬لسَّمَٰوَٰتِ وَالْأَرْضَ وَأَنزَلَ مِنَ اَ۬لسَّمَآءِ مَآءٗ فَأَخْرَجَ\nبِهِۦ مِنَ اَ۬لثَّمَرَٰتِ رِزْقاٗ لَّكُمْۖ وَسَخَّرَ لَكُمُ اُ۬لْفُلْكَ لِتَجْرِيَ\nفِے اِ۬لْبَحْرِ بِأَمْرِهِۦۖ وَسَخَّرَ لَكُمُ اُ۬لْأَنْهَٰرَۖ (34) وَسَخَّرَ لَكُمُ\nاُ۬لشَّمْسَ وَالْقَمَرَ دَآئِبَيْنِۖ وَسَخَّرَ لَكُمُ اُ۬ليْلَ وَالنَّهَارَۖ (35)\n\nوَءَاتَيٰكُم مِّن كُلِّ مَا سَأَلْتُمُوهُۖ وَإِن تَعُدُّواْ نِعْمَتَ اَ۬للَّهِ\nلَا تُحْصُوهَاۖ إِنَّ اَ۬لْإِنسَٰنَ لَظَلُومٞ كَفَّارٞۖ (36) وَإِذْ قَالَ إِبْرَٰهِيمُ\nرَبِّ اِ۪جْعَلْ هَٰذَا اَ۬لْبَلَدَ ءَامِناٗ وَاجْنُبْنِے وَبَنِيَّ أَن نَّعْبُدَ\nاَ۬لْأَصْنَامَۖ (37) رَبِّ إِنَّهُنَّ أَضْلَلْنَ كَثِيراٗ مِّنَ اَ۬لنَّاسِۖ فَمَن\nتَبِعَنِے فَإِنَّهُۥ مِنِّے وَمَنْ عَصَانِے فَإِنَّكَ غَفُورٞ رَّحِيمٞۖ (38) رَّبَّنَا\nإِنِّيَ أَسْكَنتُ مِن ذُرِّيَّتِے بِوَادٍ غَيْرِ ذِے زَرْعٍ عِندَ بَيْتِكَ\nاَ۬لْمُحَرَّمِۖ رَبَّنَا لِيُقِيمُواْ اُ۬لصَّلَوٰةَ فَاجْعَلْ أَفْـِٕدَةٗ مِّنَ اَ۬لنَّاسِ\nتَهْوِے إِلَيْهِمْ وَارْزُقْهُم مِّنَ اَ۬لثَّمَرَٰتِ لَعَلَّهُمْ يَشْكُرُونَۖ\n(39) رَبَّنَا إِنَّكَ تَعْلَمُ مَا نُخْفِے وَمَا نُعْلِنُۖ وَمَا يَخْفَيٰ عَلَي اَ۬للَّهِ\nمِن شَےْءٖ فِے اِ۬لْأَرْضِ وَلَا فِے اِ۬لسَّمَآءِۖ (40) ۞اِ۬لْحَمْدُ لِلهِ اِ۬لذِے\nوَهَبَ لِے عَلَي اَ۬لْكِبَرِ إِسْمَٰعِيلَ وَإِسْحَٰقَۖ إِنَّ رَبِّے لَسَمِيعُ\nاُ۬لدُّعَآءِۖ (41) رَبِّ اِ۪جْعَلْنِے مُقِيمَ اَ۬لصَّلَوٰةِ وَمِن ذُرِّيَّتِےۖ رَبَّنَا\nوَتَقَبَّلْ دُعَآءِۖ (42) رَبَّنَا اَ۪غْفِرْ لِے وَلِوَٰلِدَيَّ وَلِلْمُؤْمِنِينَ\nيَوْمَ يَقُومُ اُ۬لْحِسَابُۖ (43) وَلَا تَحْسِبَنَّ اَ۬للَّهَ غَٰفِلاً عَمَّا يَعْمَلُ\nاُ۬لظَّٰلِمُونَۖ إِنَّمَا يُؤَخِّرُهُمْ لِيَوْمٖ تَشْخَصُ فِيهِ اِ۬لْأَبْصَٰرُ (44)\n\nمُهْطِعِينَ مُقْنِعِے رُءُوسِهِمْ لَا يَرْتَدُّ إِلَيْهِمْ طَرْفُهُمْ\nوَأَفْـِٕدَتُهُمْ هَوَآءٞۖ (45) وَأَنذِرِ اِ۬لنَّاسَ يَوْمَ يَأْتِيهِمُ اُ۬لْعَذَابُ\nفَيَقُولُ اُ۬لذِينَ ظَلَمُواْ رَبَّنَا أَخِّرْنَا إِلَيٰ أَجَلٖ قَرِيبٖ نُّجِبْ\nدَعْوَتَكَ وَنَتَّبِعِ اِ۬لرُّسُلَۖ أَوَلَمْ تَكُونُواْ أَقْسَمْتُم مِّن قَبْلُ\nمَا لَكُم مِّن زَوَالٖۖ (46) وَسَكَنتُمْ فِے مَسَٰكِنِ اِ۬لذِينَ ظَلَمُواْ\nأَنفُسَهُمْ وَتَبَيَّنَ لَكُمْ كَيْفَ فَعَلْنَا بِهِمْ وَضَرَبْنَا لَكُمُ\nاُ۬لْأَمْثَالَۖ (47) وَقَدْ مَكَرُواْ مَكْرَهُمْ وَعِندَ اَ۬للَّهِ مَكْرُهُمْ\nوَإِن كَانَ مَكْرُهُمْ لِتَزُولَ مِنْهُ اُ۬لْجِبَالُۖ (48) فَلَا\nتَحْسِبَنَّ اَ۬للَّهَ مُخْلِفَ وَعْدِهِۦ رُسُلَهُۥۖ إِنَّ اَ۬للَّهَ عَزِيزٞ\nذُو اُ۪نتِقَامٖۖ (49) يَوْمَ تُبَدَّلُ اُ۬لْأَرْضُ غَيْرَ اَ۬لْأَرْضِ وَالسَّمَٰوَٰتُۖ\nوَبَرَزُواْ لِلهِ اِ۬لْوَٰحِدِ اِ۬لْقَهَّارِۖ (50) وَتَرَي اَ۬لْمُجْرِمِينَ يَوْمَئِذٖ\nمُّقَرَّنِينَ فِے اِ۬لْأَصْفَادِ (51) سَرَابِيلُهُم مِّن قَطِرَانٖ وَتَغْشَيٰ\nوُجُوهَهُمُ اُ۬لنَّارُ (52) لِيَجْزِيَ اَ۬للَّهُ كُلَّ نَفْسٖ مَّا كَسَبَتْۖ\nإِنَّ اَ۬للَّهَ سَرِيعُ اُ۬لْحِسَابِۖ (53) هَٰذَا بَلَٰغٞ لِّلنَّاسِ وَلِيُنذَرُواْ بِهِۦ\nوَلِيَعْلَمُواْ أَنَّمَا هُوَ إِلَٰهٞ وَٰحِدٞ وَلِيَذَّكَّرَ أُوْلُواْ اُ۬لْأَلْبَٰبِۖ (54)"
