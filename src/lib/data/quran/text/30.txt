"__BASMALAH__\n__SURAH_30__\nأَلَٓمِّٓۖ غُلِبَتِ اِ۬لرُّومُ فِے أَدْنَي اَ۬لْأَرْضِ وَهُم مِّنۢ\nبَعْدِ غَلَبِهِمْ سَيَغْلِبُونَ (1) فِے بِضْعِ سِنِينَۖ (2) لِلهِ اِ۬لْأَمْرُ\nمِن قَبْلُ وَمِنۢ بَعْدُۖ وَيَوْمَئِذٖ يَفْرَحُ اُ۬لْمُؤْمِنُونَ (3)\nبِنَصْرِ اِ۬للَّهِۖ يَنصُرُ مَنْ يَّشَآءُۖ وَهْوَ اَ۬لْعَزِيزُ اُ۬لرَّحِيمُۖ (4)\n\nوَعْدَ اَ۬للَّهِ لَا يُخْلِفُ اُ۬للَّهُ وَعْدَهُۥ وَلَٰكِنَّ أَكْثَرَ اَ۬لنَّاسِ لَا يَعْلَمُونَۖ\n(5) يَعْلَمُونَ ظَٰهِراٗ مِّنَ اَ۬لْحَيَوٰةِ اِ۬لدُّنْيَا وَهُمْ عَنِ اِ۬لْأٓخِرَةِ هُمْ\nغَٰفِلُونَۖ (6) أَوَلَمْ يَتَفَكَّرُواْ فِے أَنفُسِهِمۖ مَّا خَلَقَ اَ۬للَّهُ اُ۬لسَّمَٰوَٰتِ\nوَالْأَرْضَ وَمَا بَيْنَهُمَا إِلَّا بِالْحَقِّ وَأَجَلٖ مُّسَمّيٗۖ وَإِنَّ كَثِيراٗ\nمِّنَ اَ۬لنَّاسِ بِلِقَآءِ رَبِّهِمْ لَكَٰفِرُونَۖ (7) ۞أَوَلَمْ يَسِيرُواْ فِے\nاِ۬لْأَرْضِ فَيَنظُرُواْ كَيْفَ كَانَ عَٰقِبَةُ اُ۬لذِينَ مِن قَبْلِهِمْ كَانُواْ\nأَشَدَّ مِنْهُمْ قُوَّةٗ وَأَثَارُواْ اُ۬لْأَرْضَۖ وَعَمَرُوهَا أَكْثَرَ مِمَّا\nعَمَرُوهَا وَجَآءَتْهُمْ رُسُلُهُم بِالْبَيِّنَٰتِۖ فَمَا كَانَ اَ۬للَّهُ\nلِيَظْلِمَهُمْ وَلَٰكِن كَانُواْ أَنفُسَهُمْ يَظْلِمُونَۖ (8) ثُمَّ كَانَ\nعَٰقِبَةُ اُ۬لذِينَ أَسَٰٓـُٔواْ اُ۬لسُّوٓأَيٰ أَن كَذَّبُواْ بِـَٔايَٰتِ اِ۬للَّهِ وَكَانُواْ\nبِهَا يَسْتَهْزِءُونَۖ (9) اَ۬للَّهُ يَبْدَؤُاْ اُ۬لْخَلْقَ ثُمَّ يُعِيدُهُۥ ثُمَّ إِلَيْهِ تُرْجَعُونَۖ\n(10) وَيَوْمَ تَقُومُ اُ۬لسَّاعَةُ يُبْلِسُ اُ۬لْمُجْرِمُونَ (11) وَلَمْ يَكُن لَّهُم مِّن\nشُرَكَآئِهِمْ شُفَعَٰٓؤُاْ وَكَانُواْ بِشُرَكَآئِهِمْ كَٰفِرِينَۖ\n(12) وَيَوْمَ تَقُومُ اُ۬لسَّاعَةُ يَوْمَئِذٖ يَتَفَرَّقُونَۖ (13) فَأَمَّا اَ۬لذِينَ\nءَامَنُواْ وَعَمِلُواْ اُ۬لصَّٰلِحَٰتِ فَهُمْ فِے رَوْضَةٖ يُحْبَرُونَۖ (14)\n\nوَأَمَّا اَ۬لذِينَ كَفَرُواْ وَكَذَّبُواْ بِـَٔايَٰتِنَا وَلِقَآءِ اِ۬لْأٓخِرَةِ\nفَأُوْلَٰٓئِكَ فِے اِ۬لْعَذَابِ مُحْضَرُونَۖ (15) فَسُبْحَٰنَ اَ۬للَّهِ حِينَ تُمْسُونَ\nوَحِينَ تُصْبِحُونَ (16) وَلَهُ اُ۬لْحَمْدُ فِے اِ۬لسَّمَٰوَٰتِ وَالْأَرْضِ\nوَعَشِيّاٗ وَحِينَ تُظْهِرُونَۖ (17) يُخْرِجُ اُ۬لْحَيَّ مِنَ اَ۬لْمَيِّتِ وَيُخْرِجُ\nاُ۬لْمَيِّتَ مِنَ اَ۬لْحَيِّ وَيُحْيِ اِ۬لْأَرْضَ بَعْدَ مَوْتِهَاۖ وَكَذَٰلِكَ تُخْرَجُونَۖ\n(18) وَمِنْ ءَايَٰتِهِۦ أَنْ خَلَقَكُم مِّن تُرَابٖ ثُمَّ إِذَا أَنتُم بَشَرٞ\nتَنتَشِرُونَۖ (19) وَمِنْ ءَايَٰتِهِۦ أَنْ خَلَقَ لَكُم مِّنْ أَنفُسِكُمْ\nأَزْوَٰجاٗ لِّتَسْكُنُواْ إِلَيْهَا وَجَعَلَ بَيْنَكُم مَّوَدَّةٗ وَرَحْمَةًۖ\nإِنَّ فِے ذَٰلِكَ لَأٓيَٰتٖ لِّقَوْمٖ يَتَفَكَّرُونَۖ (20) ۞وَمِنْ ءَايَٰتِهِۦ\nخَلْقُ اُ۬لسَّمَٰوَٰتِ وَالْأَرْضِ وَاخْتِلَٰفُ أَلْسِنَتِكُمْ وَأَلْوَٰنِكُمْۖ\nإِنَّ فِے ذَٰلِكَ لَأٓيَٰتٖ لِّلْعَٰلَمِينَۖ (21) وَمِنْ ءَايَٰتِهِۦ مَنَامُكُم\nبِاليْلِ وَالنَّهَارِ وَابْتِغَآؤُكُم مِّن فَضْلِهِۦۖ إِنَّ فِے ذَٰلِكَ\nلَأٓيَٰتٖ لِّقَوْمٖ يَسْمَعُونَۖ (22) وَمِنْ ءَايَٰتِهِۦ يُرِيكُمُ اُ۬لْبَرْقَ\nخَوْفاٗ وَطَمَعاٗ وَيُنَزِّلُ مِنَ اَ۬لسَّمَآءِ مَآءٗ فَيُحْيِۦ بِهِ اِ۬لْأَرْضَ\nبَعْدَ مَوْتِهَاۖ إِنَّ فِے ذَٰلِكَ لَأٓيَٰتٖ لِّقَوْمٖ يَعْقِلُونَۖ (23)\n\nوَمِنْ ءَايَٰتِهِۦ أَن تَقُومَ اَ۬لسَّمَآءُ وَالْأَرْضُ بِأَمْرِهِۦۖ ثُمَّ إِذَا دَعَاكُمْ\nدَعْوَةٗ مِّنَ اَ۬لْأَرْضِ إِذَا أَنتُمْ تَخْرُجُونَۖ (24) وَلَهُۥ مَن فِے اِ۬لسَّمَٰوَٰتِ\nوَالْأَرْضِۖ كُلّٞ لَّهُۥ قَٰنِتُونَۖ (25) وَهْوَ اَ۬لذِے يَبْدَؤُاْ اُ۬لْخَلْقَ ثُمَّ\nيُعِيدُهُۥ وَهْوَ أَهْوَنُ عَلَيْهِۖ وَلَهُ اُ۬لْمَثَلُ اُ۬لْأَعْلَيٰ فِے اِ۬لسَّمَٰوَٰتِ\nوَالْأَرْضِۖ وَهْوَ اَ۬لْعَزِيزُ اُ۬لْحَكِيمُۖ (26) ضَرَبَ لَكُم مَّثَلاٗ\nمِّنْ أَنفُسِكُمْ هَل لَّكُم مِّن مَّا مَلَكَتْ أَيْمَٰنُكُم مِّن\nشُرَكَآءَ فِے مَا رَزَقْنَٰكُمْ فَأَنتُمْ فِيهِ سَوَآءٞ تَخَافُونَهُمْ\nكَخِيفَتِكُمْ أَنفُسَكُمْۖ كَذَٰلِكَ نُفَصِّلُ اُ۬لْأٓيَٰتِ لِقَوْمٖ\nيَعْقِلُونَۖ (27) بَلِ اِ۪تَّبَعَ اَ۬لذِينَ ظَلَمُواْ أَهْوَآءَهُم بِغَيْرِ عِلْمٖۖ\nفَمَنْ يَّهْدِے مَنْ أَضَلَّ اَ۬للَّهُۖ وَمَا لَهُم مِّن نَّٰصِرِينَۖ (28) فَأَقِمْ\nوَجْهَكَ لِلدِّينِ حَنِيفاٗ فِطْرَتَ اَ۬للَّهِ اِ۬لتِے فَطَرَ اَ۬لنَّاسَ عَلَيْهَا\nلَا تَبْدِيلَ لِخَلْقِ اِ۬للَّهِۖ ذَٰلِكَ اَ۬لدِّينُ اُ۬لْقَيِّمُ وَلَٰكِنَّ أَكْثَرَ\nاَ۬لنَّاسِ لَا يَعْلَمُونَ (29) ۞مُنِيبِينَ إِلَيْهِ وَاتَّقُوهُ وَأَقِيمُواْ\nاُ۬لصَّلَوٰةَ وَلَا تَكُونُواْ مِنَ اَ۬لْمُشْرِكِينَ (30) مِنَ اَ۬لذِينَ فَرَّقُواْ\nدِينَهُمْ وَكَانُواْ شِيَعاٗۖ كُلُّ حِزْبِۢ بِمَا لَدَيْهِمْ فَرِحُونَۖ (31)\n\nوَإِذَا مَسَّ اَ۬لنَّاسَ ضُرّٞ دَعَوْاْ رَبَّهُم مُّنِيبِينَ إِلَيْهِ ثُمَّ إِذَا أَذَاقَهُم\nمِّنْهُ رَحْمَةً إِذَا فَرِيقٞ مِّنْهُم بِرَبِّهِمْ يُشْرِكُونَۖ (32) لِيَكْفُرُواْ بِمَا\nءَاتَيْنَٰهُمْۖ فَتَمَتَّعُواْ فَسَوْفَ تَعْلَمُونَۖ (33) أَمْ أَنزَلْنَا عَلَيْهِمْ\nسُلْطَٰناٗ فَهْوَ يَتَكَلَّمُ بِمَا كَانُواْ بِهِۦ يُشْرِكُونَۖ (34) وَإِذَا أَذَقْنَا\nاَ۬لنَّاسَ رَحْمَةٗ فَرِحُواْ بِهَاۖ وَإِن تُصِبْهُمْ سَيِّئَةُۢ بِمَا قَدَّمَتْ أَيْدِيهِمْ\nإِذَا هُمْ يَقْنَطُونَۖ (35) أَوَلَمْ يَرَوْاْ أَنَّ اَ۬للَّهَ يَبْسُطُ اُ۬لرِّزْقَ لِمَنْ يَّشَآءُ\nوَيَقْدِرُۖ إِنَّ فِے ذَٰلِكَ لَأٓيَٰتٖ لِّقَوْمٖ يُؤْمِنُونَۖ (36) فَـَٔاتِ ذَا اَ۬لْقُرْبَيٰ\nحَقَّهُۥ وَالْمِسْكِينَ وَابْنَ اَ۬لسَّبِيلِۖ ذَٰلِكَ خَيْرٞ لِّلذِينَ يُرِيدُونَ\nوَجْهَ اَ۬للَّهِۖ وَأُوْلَٰٓئِكَ هُمُ اُ۬لْمُفْلِحُونَۖ (37) وَمَا ءَاتَيْتُم مِّن رِّباٗ\nلِّتُرْبُواْ فِے أَمْوَٰلِ اِ۬لنَّاسِ فَلَا يَرْبُواْ عِندَ اَ۬للَّهِۖ وَمَا ءَاتَيْتُم مِّن\nزَكَوٰةٖ تُرِيدُونَ وَجْهَ اَ۬للَّهِ فَأُوْلَٰٓئِكَ هُمُ اُ۬لْمُضْعِفُونَۖ (38)\nاَ۬للَّهُ اُ۬لذِے خَلَقَكُمْ ثُمَّ رَزَقَكُمْ ثُمَّ يُمِيتُكُمْ ثُمَّ يُحْيِيكُمْۖ هَلْ مِن\nشُرَكَآئِكُم مَّنْ يَّفْعَلُ مِن ذَٰلِكُم مِّن شَےْءٖۖ سُبْحَٰنَهُۥ وَتَعَٰلَيٰ\nعَمَّا يُشْرِكُونَۖ (39) ۞ظَهَرَ اَ۬لْفَسَادُ فِے اِ۬لْبَرِّ وَالْبَحْرِ بِمَا كَسَبَتْ\nأَيْدِے اِ۬لنَّاسِ لِيُذِيقَهُم بَعْضَ اَ۬لذِے عَمِلُواْ لَعَلَّهُمْ يَرْجِعُونَۖ (40)\n\nقُلْ سِيرُواْ فِے اِ۬لْأَرْضِ فَانظُرُواْ كَيْفَ كَانَ عَٰقِبَةُ اُ۬لذِينَ مِن قَبْلُۖ\nكَانَ أَكْثَرُهُم مُّشْرِكِينَۖ (41) فَأَقِمْ وَجْهَكَ لِلدِّينِ اِ۬لْقَيِّمِ مِن\nقَبْلِ أَنْ يَّأْتِيَ يَوْمٞ لَّا مَرَدَّ لَهُۥ مِنَ اَ۬للَّهِۖ يَوْمَئِذٖ يَصَّدَّعُونَۖ (42) مَن\nكَفَرَ فَعَلَيْهِ كُفْرُهُۥۖ وَمَنْ عَمِلَ صَٰلِحاٗ فَلِأَنفُسِهِمْ يَمْهَدُونَ (43)\nلِيَجْزِيَ اَ۬لذِينَ ءَامَنُواْ وَعَمِلُواْ اُ۬لصَّٰلِحَٰتِ مِن فَضْلِهِۦۖ إِنَّهُۥ لَا يُحِبُّ\nاُ۬لْكَٰفِرِينَۖ (44) وَمِنْ ءَايَٰتِهِۦ أَنْ يُّرْسِلَ اَ۬لرِّيَاحَ مُبَشِّرَٰتٖ وَلِيُذِيقَكُم\nمِّن رَّحْمَتِهِۦ وَلِتَجْرِيَ اَ۬لْفُلْكُ بِأَمْرِهِۦ وَلِتَبْتَغُواْ مِن فَضْلِهِۦ وَلَعَلَّكُمْ\nتَشْكُرُونَۖ (45) وَلَقَدْ أَرْسَلْنَا مِن قَبْلِكَ رُسُلاً إِلَيٰ قَوْمِهِمْ فَجَآءُوهُم\nبِالْبَيِّنَٰتِ فَانتَقَمْنَا مِنَ اَ۬لذِينَ أَجْرَمُواْۖ وَكَانَ حَقّاً عَلَيْنَا نَصْرُ\nاُ۬لْمُؤْمِنِينَۖ (46) اَ۬للَّهُ اُ۬لذِے يُرْسِلُ اُ۬لرِّيَٰحَ فَتُثِيرُ سَحَاباٗ فَيَبْسُطُهُۥ\nفِے اِ۬لسَّمَآءِ كَيْفَ يَشَآءُ وَيَجْعَلُهُۥ كِسَفاٗ فَتَرَي اَ۬لْوَدْقَ يَخْرُجُ مِنْ\nخِلَٰلِهِۦۖ فَإِذَا أَصَابَ بِهِۦ مَنْ يَّشَآءُ مِنْ عِبَادِهِۦ إِذَا هُمْ يَسْتَبْشِرُونَۖ\n(47) وَإِن كَانُواْ مِن قَبْلِ أَنْ يُّنَزَّلَ عَلَيْهِم مِّن قَبْلِهِۦ لَمُبْلِسِينَۖ\n(48) فَانظُرْ إِلَيٰ أَثَرِ رَحْمَتِ اِ۬للَّهِ كَيْفَ يُحْيِ اِ۬لْأَرْضَ بَعْدَ مَوْتِهَاۖ\nإِنَّ ذَٰلِكَ لَمُحْيِ اِ۬لْمَوْتَيٰۖ وَهْوَ عَلَيٰ كُلِّ شَےْءٖ قَدِيرٞۖ (49)\n\nوَلَئِنْ أَرْسَلْنَا رِيحاٗ فَرَأَوْهُ مُصْفَرّاٗ لَّظَلُّواْ مِنۢ بَعْدِهِۦ يَكْفُرُونَۖ\n(50) فَإِنَّكَ لَا تُسْمِعُ اُ۬لْمَوْتَيٰ وَلَا تُسْمِعُ اُ۬لصُّمَّ اَ۬لدُّعَآءَ ا۪ذَا وَلَّوْاْ\nمُدْبِرِينَۖ (51) وَمَا أَنتَ بِهَٰدِ اِ۬لْعُمْيِ عَن ضَلَٰلَتِهِمْۖ إِن تُسْمِعُ إِلَّا\nمَنْ يُّؤْمِنُ بِـَٔايَٰتِنَا فَهُم مُّسْلِمُونَۖ (52) ۞اَ۬للَّهُ اُ۬لذِے خَلَقَكُم\nمِّن ضُعْفٖ ثُمَّ جَعَلَ مِنۢ بَعْدِ ضُعْفٖ قُوَّةٗ ثُمَّ جَعَلَ مِنۢ بَعْدِ\nقُوَّةٖ ضُعْفاٗ وَشَيْبَةٗۖ يَخْلُقُ مَا يَشَآءُۖ وَهْوَ اَ۬لْعَلِيمُ اُ۬لْقَدِيرُۖ\n(53) وَيَوْمَ تَقُومُ اُ۬لسَّاعَةُ يُقْسِمُ اُ۬لْمُجْرِمُونَ مَا لَبِثُواْ غَيْرَ\nسَاعَةٖۖ كَذَٰلِكَ كَانُواْ يُؤْفَكُونَۖ (54) وَقَالَ اَ۬لذِينَ أُوتُواْ\nاُ۬لْعِلْمَ وَالْإِيمَٰنَ لَقَدْ لَبِثْتُمْ فِے كِتَٰبِ اِ۬للَّهِ إِلَيٰ يَوْمِ اِ۬لْبَعْثِۖ\nفَهَٰذَا يَوْمُ اُ۬لْبَعْثِ وَلَٰكِنَّكُمْ كُنتُمْ لَا تَعْلَمُونَۖ (55) فَيَوْمَئِذٖ\nلَّا تَنفَعُ اُ۬لذِينَ ظَلَمُواْ مَعْذِرَتُهُمْ وَلَا هُمْ يُسْتَعْتَبُونَۖ\n(56) وَلَقَدْ ضَرَبْنَا لِلنَّاسِ فِے هَٰذَا اَ۬لْقُرْءَانِ مِن كُلِّ مَثَلٖۖ\nوَلَئِن جِئْتَهُم بِـَٔايَةٖ لَّيَقُولَنَّ اَ۬لذِينَ كَفَرُواْ إِنْ أَنتُمْ إِلَّا\nمُبْطِلُونَۖ (57) كَذَٰلِكَ يَطْبَعُ اُ۬للَّهُ عَلَيٰ قُلُوبِ اِ۬لذِينَ لَا يَعْلَمُونَۖ\n(58) فَاصْبِرْ إِنَّ وَعْدَ اَ۬للَّهِ حَقّٞۖ وَلَا يَسْتَخِفَّنَّكَ اَ۬لذِينَ لَا يُوقِنُونَۖ (59)"
