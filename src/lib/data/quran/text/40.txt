"حَمِٓۖ تَنزِيلُ اُ۬لْكِتَٰبِ مِنَ اَ۬للَّهِ اِ۬لْعَزِيزِ اِ۬لْعَلِيمِ (1) غَافِرِ اِ۬لذَّنۢبِ\nوَقَابِلِ اِ۬لتَّوْبِ شَدِيدِ اِ۬لْعِقَابِ ذِے اِ۬لطَّوْلِۖ لَا إِلَٰهَ إِلَّا هُوَۖ إِلَيْهِ\nاِ۬لْمَصِيرُۖ (2) مَا يُجَٰدِلُ فِے ءَايَٰتِ اِ۬للَّهِ إِلَّا اَ۬لذِينَ كَفَرُواْ فَلَا يَغْرُرْكَ\nتَقَلُّبُهُمْ فِے اِ۬لْبِلَٰدِۖ (3) كَذَّبَتْ قَبْلَهُمْ قَوْمُ نُوحٖ وَالْأَحْزَابُ\nمِنۢ بَعْدِهِمْۖ وَهَمَّتْ كُلُّ أُمَّةِۢ بِرَسُولِهِمْ لِيَأْخُذُوهُۖ\nوَجَٰدَلُواْ بِالْبَٰطِلِ لِيُدْحِضُواْ بِهِ اِ۬لْحَقَّ فَأَخَذتُّهُمْۖ فَكَيْفَ\nكَانَ عِقَابِۖ (4) وَكَذَٰلِكَ حَقَّتْ كَلِمَٰتُ رَبِّكَ عَلَي اَ۬لذِينَ\nكَفَرُواْ أَنَّهُمْ أَصْحَٰبُ اُ۬لنَّارِۖ (5) اِ۬لذِينَ يَحْمِلُونَ اَ۬لْعَرْشَ\nوَمَنْ حَوْلَهُۥ يُسَبِّحُونَ بِحَمْدِ رَبِّهِمْ وَيُؤْمِنُونَ بِهِۦ وَيَسْتَغْفِرُونَ\nلِلذِينَ ءَامَنُواْۖ رَبَّنَا وَسِعْتَ كُلَّ شَےْءٖ رَّحْمَةٗ وَعِلْماٗ فَاغْفِرْ\nلِلذِينَ تَابُواْ وَاتَّبَعُواْ سَبِيلَكَ وَقِهِمْ عَذَابَ اَ۬لْجَحِيمِۖ (6)\n\n رَبَّنَا وَأَدْخِلْهُمْ جَنَّٰتِ عَدْنٍ اِ۬لتِے وَعَدتَّهُمْ وَمَن صَلَحَ\nمِنْ ءَابَآئِهِمْ وَأَزْوَٰجِهِمْ وَذُرِّيَّٰتِهِمْۖ إِنَّكَ أَنتَ اَ۬لْعَزِيزُ\nاُ۬لْحَكِيمُۖ (7) وَقِهِمُ اُ۬لسَّيِّـَٔاتِۖ وَمَن تَقِ اِ۬لسَّيِّـَٔاتِ\nيَوْمَئِذٖ فَقَدْ رَحِمْتَهُۥۖ وَذَٰلِكَ هُوَ اَ۬لْفَوْزُ اُ۬لْعَظِيمُۖ (8) ۞إِنَّ\nاَ۬لذِينَ كَفَرُواْ يُنَادَوْنَ لَمَقْتُ اُ۬للَّهِ أَكْبَرُ مِن مَّقْتِكُمْ\nأَنفُسَكُمْ إِذْ تُدْعَوْنَ إِلَي اَ۬لْإِيمَٰنِ فَتَكْفُرُونَۖ (9) قَالُواْ رَبَّنَا\nأَمَتَّنَا اَ۪ثْنَتَيْنِ وَأَحْيَيْتَنَا اَ۪ثْنَتَيْنِ فَاعْتَرَفْنَا بِذُنُوبِنَا فَهَلْ\nإِلَيٰ خُرُوجٖ مِّن سَبِيلٖۖ (10) ذَٰلِكُم بِأَنَّهُۥ إِذَا دُعِيَ اَ۬للَّهُ\nوَحْدَهُۥ كَفَرْتُمْ وَإِنْ يُّشْرَكْ بِهِۦ تُؤْمِنُواْۖ فَالْحُكْمُ لِلهِ\nاِ۬لْعَلِيِّ اِ۬لْكَبِيرِۖ (11) هُوَ اَ۬لذِے يُرِيكُمْ ءَايَٰتِهِۦ وَيُنَزِّلُ لَكُم مِّنَ\nاَ۬لسَّمَآءِ رِزْقاٗۖ وَمَا يَتَذَكَّرُ إِلَّا مَنْ يُّنِيبُۖ (12) فَادْعُواْ اُ۬للَّهَ\nمُخْلِصِينَ لَهُ اُ۬لدِّينَ وَلَوْ كَرِهَ اَ۬لْكَٰفِرُونَۖ (13) رَفِيعُ\nاُ۬لدَّرَجَٰتِ ذُو اُ۬لْعَرْشِۖ يُلْقِے اِ۬لرُّوحَ مِنْ أَمْرِهِۦ عَلَيٰ مَنْ يَّشَآءُ\nمِنْ عِبَادِهِۦ لِيُنذِرَ يَوْمَ اَ۬لتَّلَٰقِ (14) يَوْمَ هُم بَٰرِزُونَ لَا يَخْفَيٰ\nعَلَي اَ۬للَّهِ مِنْهُمْ شَےْءٞۖ لِّمَنِ اِ۬لْمُلْكُ اُ۬لْيَوْمَۖ لِلهِ اِ۬لْوَٰحِدِ اِ۬لْقَهَّارِۖ (15)\n\nاِ۬لْيَوْمَ تُجْزَيٰ كُلُّ نَفْسِۢ بِمَا كَسَبَتْۖ لَا ظُلْمَ اَ۬لْيَوْمَۖ إِنَّ\nاَ۬للَّهَ سَرِيعُ اُ۬لْحِسَابِۖ (16) وَأَنذِرْهُمْ يَوْمَ اَ۬لْأٓزِفَةِ إِذِ اِ۬لْقُلُوبُ\nلَدَي اَ۬لْحَنَاجِرِ كَٰظِمِينَۖ (17) مَا لِلظَّٰلِمِينَ مِنْ حَمِيمٖ وَلَا شَفِيعٖ\nيُطَاعُۖ (18) يَعْلَمُ خَآئِنَةَ اَ۬لْأَعْيُنِ وَمَا تُخْفِے اِ۬لصُّدُورُۖ (19) وَاللَّهُ\nيَقْضِے بِالْحَقِّۖ وَالذِينَ تَدْعُونَ مِن دُونِهِۦ لَا يَقْضُونَ\nبِشَےْءٍۖ إِنَّ اَ۬للَّهَ هُوَ اَ۬لسَّمِيعُ اُ۬لْبَصِيرُۖ (20) ۞أَوَلَمْ يَسِيرُواْ فِے\nاِ۬لْأَرْضِ فَيَنظُرُواْ كَيْفَ كَانَ عَٰقِبَةُ اُ۬لذِينَ كَانُواْ مِن قَبْلِهِمْۖ\nكَانُواْ هُمْ أَشَدَّ مِنْهُمْ قُوَّةٗ وَءَاثَاراٗ فِے اِ۬لْأَرْضِ فَأَخَذَهُمُ اُ۬للَّهُ\nبِذُنُوبِهِمْۖ وَمَا كَانَ لَهُم مِّنَ اَ۬للَّهِ مِنْ وَّاقٖۖ (21) ذَٰلِكَ بِأَنَّهُمْ\nكَانَت تَّأْتِيهِمْ رُسُلُهُم بِالْبَيِّنَٰتِ فَكَفَرُواْ فَأَخَذَهُمُ اُ۬للَّهُۖ\nإِنَّهُۥ قَوِيّٞ شَدِيدُ اُ۬لْعِقَابِۖ (22) وَلَقَدْ أَرْسَلْنَا مُوسَيٰ بِـَٔايَٰتِنَا\nوَسُلْطَٰنٖ مُّبِينٍ (23) إِلَيٰ فِرْعَوْنَ وَهَامَٰنَ وَقَارُونَ\nفَقَالُواْ سَٰحِرٞ كَذَّابٞۖ (24) فَلَمَّا جَآءَهُم بِالْحَقِّ مِنْ\nعِندِنَا قَالُواْ اُ۟قْتُلُواْ أَبْنَآءَ اَ۬لذِينَ ءَامَنُواْ مَعَهُۥ وَاسْتَحْيُواْ\nنِسَآءَهُمْۖ وَمَا كَيْدُ اُ۬لْكَٰفِرِينَ إِلَّا فِے ضَلَٰلٖۖ (25)\n\nوَقَالَ فِرْعَوْنُ ذَرُونِے أَقْتُلْ مُوسَيٰ وَلْيَدْعُ رَبَّهُۥ إِنِّيَ أَخَافُ\nأَنْ يُّبَدِّلَ دِينَكُمْ وَأَنْ يُّظْهِرَ فِے اِ۬لْأَرْضِ اِ۬لْفَسَادَۖ (26)\nوَقَالَ مُوسَيٰ إِنِّے عُذْتُ بِرَبِّے وَرَبِّكُم مِّن كُلِّ مُتَكَبِّرٖ لَّا يُؤْمِنُ\nبِيَوْمِ اِ۬لْحِسَابِۖ (27) وَقَالَ رَجُلٞ مُّؤْمِنٞ مِّنْ ءَالِ فِرْعَوْنَ\nيَكْتُمُ إِيمَٰنَهُۥ أَتَقْتُلُونَ رَجُلاً أَنْ يَّقُولَ رَبِّيَ اَ۬للَّهُ وَقَدْ\nجَآءَكُم بِالْبَيِّنَٰتِ مِن رَّبِّكُمْۖ وَإِنْ يَّكُ كَٰذِباٗ فَعَلَيْهِ\nكَذِبُهُۥۖ وَإِنْ يَّكُ صَادِقاٗ يُصِبْكُم بَعْضُ اُ۬لذِے يَعِدُكُمْۖ\nإِنَّ اَ۬للَّهَ لَا يَهْدِے مَنْ هُوَ مُسْرِفٞ كَذَّابٞۖ (28) يَٰقَوْمِ لَكُمُ\nاُ۬لْمُلْكُ اُ۬لْيَوْمَ ظَٰهِرِينَ فِے اِ۬لْأَرْضِ فَمَنْ يَّنصُرُنَا مِنۢ بَأْسِ اِ۬للَّهِ\nإِن جَآءَنَاۖ قَالَ فِرْعَوْنُ مَا أُرِيكُمْ إِلَّا مَا أَرَيٰ وَمَا أَهْدِيكُمْ\nإِلَّا سَبِيلَ اَ۬لرَّشَادِۖ (29) ۞وَقَالَ اَ۬لذِے ءَامَنَ يَٰقَوْمِ إِنِّيَ أَخَافُ\nعَلَيْكُم مِّثْلَ يَوْمِ اِ۬لْأَحْزَابِ (30) مِثْلَ دَأْبِ قَوْمِ نُوحٖ وَعَادٖ\nوَثَمُودَ وَالذِينَ مِنۢ بَعْدِهِمْۖ وَمَا اَ۬للَّهُ يُرِيدُ ظُلْماٗ لِّلْعِبَادِۖ (31)\nوَيَٰقَوْمِ إِنِّيَ أَخَافُ عَلَيْكُمْ يَوْمَ اَ۬لتَّنَادِ (32) يَوْمَ تُوَلُّونَ مُدْبِرِينَ\nمَا لَكُم مِّنَ اَ۬للَّهِ مِنْ عَٰصِمٖۖ وَمَنْ يُّضْلِلِ اِ۬للَّهُ فَمَا لَهُۥ مِنْ هَادٖۖ كَذَٰلِكَ يُضِلُّ اُ۬للَّهُ مَنْ هُوَ مُسْرِفٞ\nمُّرْتَابٌۖ (34) اِ۬لذِينَ يُجَٰدِلُونَ فِے ءَايَٰتِ اِ۬للَّهِ بِغَيْرِ سُلْطَٰنٍ\nأَتَيٰهُمْۖ كَبُرَ مَقْتاً عِندَ اَ۬للَّهِ وَعِندَ اَ۬لذِينَ مُّغْنُونَ عَنَّا نَصِيباٗ مِّنَ اَ۬لنَّارِۖ (47) قَالَ\nاَ۬لذِينَ اَ۪سْتَكْبَرُواْ إِنَّا كُلّٞ فِيهَا إِنَّ اَ۬للَّهَ قَدْ حَكَمَ\nبَيْنَ اَ۬لْعِبَادِۖ (48) وَقَالَ اَ۬لذِينَ فِے اِ۬لنَّارِ لِخَزَنَةِ جَهَنَّمَ\nاَ۟دْعُواْ رَبَّكُمْ يُخَفِّفْ عَنَّا يَوْماٗ مِّنَ اَ۬لْعَذَابِۖ (49)\n\nقَالُواْ أَوَلَمْ تَكُ تَأْتِيكُمْ رُسُلُكُم بِالْبَيِّنَٰتِ قَالُواْ بَلَيٰ\nقَالُواْ فَادْعُواْۖ وَمَا دُعَٰٓؤُاْ اُ۬لْكَٰفِرِينَ إِلَّا فِے ضَلَٰلٍۖ (50)\nإِنَّا لَنَنصُرُ رُسُلَنَا وَالذِينَ ءَامَنُواْ فِے اِ۬لْحَيَوٰةِ اِ۬لدُّنْيَا\nوَيَوْمَ يَقُومُ اُ۬لْأَشْهَٰدُ (51) يَوْمَ لَا يَنفَعُ اُ۬لظَّٰلِمِينَ مَعْذِرَتُهُمْ\nوَلَهُمُ اُ۬للَّعْنَةُ وَلَهُمْ سُوٓءُ اُ۬لدَّارِۖ (52) ۞وَلَقَدْ ءَاتَيْنَا مُوسَي\nاَ۬لْهُدَيٰ وَأَوْرَثْنَا بَنِے إِسْرَآءِيلَ اَ۬لْكِتَٰبَ هُديٗ\nوَذِكْرَيٰ لِأُوْلِے اِ۬لْأَلْبَٰبِۖ (53) فَاصْبِرْ إِنَّ وَعْدَ اَ۬للَّهِ\nحَقّٞ وَاسْتَغْفِرْ لِذَنۢبِكَ وَسَبِّحْ بِحَمْدِ رَبِّكَ بِالْعَشِيِّ\nوَالْإِبْكَٰرِۖ (54) إِنَّ اَ۬لذِينَ يُجَٰدِلُونَ فِے ءَايَٰتِ اِ۬للَّهِ\nبِغَيْرِ سُلْطَٰنٍ أَتَيٰهُمْ إِن فِے صُدُورِهِمْ إِلَّا كِبْرٞ\nمَّا هُم بِبَٰلِغِيهِۖ فَاسْتَعِذْ بِاللَّهِۖ إِنَّهُۥ هُوَ اَ۬لسَّمِيعُ\nاُ۬لْبَصِيرُۖ (55) لَخَلْقُ اُ۬لسَّمَٰوَٰتِ وَالْأَرْضِ أَكْبَرُ مِنْ\nخَلْقِ اِ۬لنَّاسِ وَلَٰكِنَّ أَكْثَرَ اَ۬لنَّاسِ لَا يَعْلَمُونَۖ (56)\nوَمَا يَسْتَوِے اِ۬لْأَعْمَيٰ وَالْبَصِيرُ (57) وَالذِينَ ءَامَنُواْ وَعَمِلُواْ\nاُ۬لصَّٰلِحَٰتِ وَلَا اَ۬لْمُسِےٓءُۖ قَلِيلاٗ مَّا يَتَذَكَّرُونَۖ (58)\n\nإِنَّ اَ۬لسَّاعَةَ لَأٓتِيَةٞ لَّا رَيْبَ فِيهَا وَلَٰكِنَّ أَكْثَرَ اَ۬لنَّاسِ\nلَا يُؤْمِنُونَۖ (59) وَقَالَ رَبُّكُمُ اُ۟دْعُونِے أَسْتَجِبْ لَكُمْۖ\nإِنَّ اَ۬لذِينَ يَسْتَكْبِرُونَ عَنْ عِبَادَتِے سَيَدْخُلُونَ جَهَنَّمَ\nدَاخِرِينَۖ (60) اَ۬للَّهُ اُ۬لذِے جَعَلَ لَكُمُ اُ۬ليْلَ لِتَسْكُنُواْ\nفِيهِ وَالنَّهَارَ مُبْصِراًۖ إِنَّ اَ۬للَّهَ لَذُو فَضْلٍ عَلَي اَ۬لنَّاسِ وَلَٰكِنَّ\nأَكْثَرَ اَ۬لنَّاسِ لَا يَشْكُرُونَۖ (61) ذَٰلِكُمُ اُ۬للَّهُ رَبُّكُمْ\nخَٰلِقُ كُلِّ شَےْءٖ لَّا إِلَٰهَ إِلَّا هُوَۖ فَأَنَّيٰ تُؤْفَكُونَۖ وَرَزَقَكُم مِّنَ\nاَ۬لطَّيِّبَٰتِۖ ذَٰلِكُمُ اُ۬للَّهُ رَبُّكُمْۖ فَتَبَٰرَكَ اَ۬للَّهُ رَبُّ\nاُ۬لْعَٰلَمِينَۖ (64) هُوَ اَ۬لْحَيُّ لَا إِلَٰهَ إِلَّا هُوَ فَادْعُوهُ مُخْلِصِينَ\nلَهُ اُ۬لدِّينَۖ اَ۬لْحَمْدُ لِلهِ رَبِّ اِ۬لْعَٰلَمِينَۖ (65) ۞قُلْ إِنِّے\nنُهِيتُ أَنْ أَعْبُدَ اَ۬لذِينَ تَدْعُونَ مِن دُونِ اِ۬للَّهِ لَمَّا جَآءَنِيَ\nاَ۬لْبَيِّنَٰتُ مِن رَّبِّے وَأُمِرْتُ أَنْ أُسْلِمَ
